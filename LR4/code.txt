#include <iostream>
#include <cmath> // Для функції abs
using namespace std;

int main(void) {
    // Константа для epsilon порівняння
    const float EPSILON = 0.000001;

    // Запит двох цілих чисел у користувача
    int num1, num2;
    cout << "Enter two integer numbers: ";
    cin >> num1 >> num2;

    // Ділення 1 на кожне з чисел з приведенням до float
    float result1 = 1.0f / num1;
    float result2 = 1.0f / num2;

    // Порівняння результатів з використанням epsilon
    if (abs(result1 - result2) < EPSILON) {
        cout << "Results are equal (by 0.000001 epsilon)" << endl;
    } else {
        cout << "Results are not equal (by 0.000001 epsilon)" << endl;
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    int num1, num2, num3, num4;

    cout << "Enter four numbers for IP address (each between 1 and 255):" << endl;
    cin >> num1 >> num2 >> num3 >> num4;

    // Check if the numbers are within the range 1-255
    if ((num1 >= 1 && num1 <= 255) &&
        (num2 >= 1 && num2 <= 255) &&
        (num3 >= 1 && num3 <= 255) &&
        (num4 >= 1 && num4 <= 255)) {
        
        // Concatenate and print the IP address
        cout << num1 << "." << num2 << "." << num3 << "." << num4 << endl;
    } else {
        cout << "One or more numbers are out of the range 1-255." << endl;
    }

    return 0;
}
